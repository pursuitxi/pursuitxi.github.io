\end{align*}

\begin{align*}
E{+}(x,t)&=(2\pi){-3}\int_{0}{+\infty}\frac{\sin(\eta t)}{\eta}\eta^2 d\eta\int_{0}{2\pi}d\varphi\int_{0}{\pi}e^{i|x|\eta \cos(\theta)}\sin\theta d\theta\
&=(2\pi){-2}\int_{0}{\infty}\frac{\sin(\eta t)}{|x|}d\eta\int_{-1}^{1}\cos(|x|\eta y)+i\sin(|x|\eta y)d(|x|\eta y)\
&=\frac{2}{4\pi2|x|}\int_{0}{+\infty}\sin(\eta t)\sin(|x|\eta)d\eta\
&=\frac{2}{4\pi2|x|}\cdot\frac{1}{2}\int_{0}{+\infty}\cos((|x|-t)\eta)-\cos((|x|+t)\eta)d\eta\
&=\frac{1}{4\pi^2|x|}\lim\limits_{A\to +\infty}\int_{0}^{A}\cos((|x|-t)\eta)-\cos((|x|+t)\eta)d\eta\
&=\frac{1}{4\pi|x|}\lim\limits_{A\to \infty}\frac{1}{\pi}\Big[\frac{\sin A(|x|-t)}{|x|-t}-\frac{\sin A(|x|+t)}{|x|+t}\Big]\
&=\frac{1}{4\pi|x|}(\delta(|x|-t)-\delta(|x|+t))\
&=\frac{H(t)}{4\pi |x|}\delta(|x|-t)
\end{align*}
